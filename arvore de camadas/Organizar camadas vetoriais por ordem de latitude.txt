from qgis.utils import iface

layers = [
    l for l in QgsProject.instance().mapLayers().values()
    if l.name().startswith('BH') or l.name().startswith('-BH')
]

layer_lat = []
for layer in layers:
    if layer.isSpatial():
        lat = layer.extent().center().y()
        layer_lat.append((layer, lat))

layer_lat_sorted = sorted(layer_lat, key=lambda x: x[1])

root = QgsProject.instance().layerTreeRoot()
for lyr, _ in layer_lat_sorted:
    node = root.findLayer(lyr.id())
    root.removeChildNode(node)
    root.insertChildNode(0, node)
