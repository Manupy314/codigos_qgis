root = QgsProject.instance().layerTreeRoot()
grupo = root.findGroup("MAPAS")   # <<< altere aqui

if not grupo:
    raise Exception("Grupo nÃ£o encontrado!")

for node in grupo.children():
    if hasattr(node, "layer") and node.layer():
        lyr = node.layer()
        lyr.setName(lyr.name().upper())

print("Camadas do grupo convertidas para CAIXA ALTA.")
